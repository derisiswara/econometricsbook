```{r}
# install.packages("ardl.nardl")
library(ardl.nardl)
```

```{r}
# Data
datanardl <- read.csv("Data/Bab 9/datanardl.csv")
head(datanardl)
```

```{r}
# Phillips-Perron Unit Root Test
